```py
  epsilon := Round(0.1 * Min(buf.Width, buf.Height));

  yPlaceMin := 0 + epsilon;
  yPlaceMax := buf.Height - epsilon;
  dx := (xPlaceRight - xPlaceLeft) / (xright - xleft);
  dy := (yPlaceMax - yPlaceMin) / (ymax - ymin);
  buf.Canvas.MoveTo(Round(xPlaceLeft + (src[0][Ix] - xLeft) * dx),
    Round(yPlaceMax - (src[0][Iy] - yMin) * dy));

```
